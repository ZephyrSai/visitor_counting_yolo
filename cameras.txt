# Example URLs file with per-camera model and bidirectional counting
# Format: URL|model|direction|fps
# All fields except URL are optional

# BIDIRECTIONAL COUNTING - Counts both IN and OUT
# Camera at entrance/exit - counts people going both ways
rtsp://admin:pass@192.168.1.100:554/entrance|yolo11x.pt|bidirectional_horizontal
rtsp://admin:pass@192.168.1.101:554/hallway|yolo11x.pt|bidirectional_vertical

# SINGLE DIRECTION COUNTING
# Cameras using yolo11n.pt (lightweight model for simple scenes)
rtsp://admin:pass@192.168.1.102:554/stream|yolo11n.pt|top_to_bottom
rtsp://admin:pass@192.168.1.103:554/stream|yolo11n.pt|bottom_to_top

# Mixed configurations
# Bidirectional with large model for complex entrance
rtsp://admin:pass@192.168.1.104:554/main_entrance|yolo12l.pt|bidirectional_horizontal|3

# Single direction with standard model
rtsp://admin:pass@192.168.1.105:554/stream|yolo11x.pt|left_to_right|5

# Camera with all options specified (bidirectional vertical counting)
rtsp://admin:pass@192.168.1.106:554/corridor|yolo11x.pt|bidirectional_vertical|10

# Camera using default settings (will use command line parameters)
rtsp://admin:pass@192.168.1.107:554/stream

# Another bidirectional example for store entrance
rtsp://admin:pass@192.168.1.108:554/store_door|yolo11x.pt|bidirectional_horizontal|5
